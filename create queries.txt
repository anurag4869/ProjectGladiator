CREATE TABLE TABLE_SUBJECT 
(
  SUBJECT_ID NUMBER(5) NOT NULL 
, SUBJECT_NAME VARCHAR2(20) NOT NULL 
, CONSTRAINT TABLE_SUBJECT_PK PRIMARY KEY 
  (
    SUBJECT_ID 
  )
  ENABLE 
);


CREATE TABLE TABLE_QUESTION 
(
  QUESTION_ID NUMBER(5) NOT NULL 
, QUESTION VARCHAR2(100) NOT NULL 
, CHOICE1 VARCHAR2(100) NOT NULL 
, CHOICE2 VARCHAR2(100) NOT NULL 
, CHOICE3 VARCHAR2(100) NOT NULL 
, CHOICE4 VARCHAR2(100) NOT NULL 
, SUBJECT_ID NUMBER(5) NOT NULL 
, CORRECT_ANS VARCHAR2(100) NOT NULL 
, QUESTION_LEVEL NUMBER(1) NOT NULL 
, CONSTRAINT TABLE_QUESTION_PK PRIMARY KEY 
  (
    QUESTION_ID 
  )
, CONSTRAINT TABLE_QUESTION_SUBJECT_FK FOREIGN KEY(SUBJECT_ID) REFERENCES TABLE_SUBJECT(SUBJECT_ID) 
  ENABLE 
);

CREATE TABLE TABLE_TEST 
(
  TEST_ID NUMBER(5) NOT NULL 
, STUDENT_ID NUMBER(5) NOT NULL 
, SUBJECT_ID NUMBER(5) NOT NULL 
, CONSTRAINT TABLE_TEST_PK PRIMARY KEY 
  (
    TEST_ID 
  )
  , CONSTRAINT TABLE_TEST_STUDENT_FK FOREIGN KEY(STUDENT_ID) REFERENCES TABLE_STUDENT(STUDENT_ID)
, CONSTRAINT TABLE_TEST_SUBJECT_FK FOREIGN KEY(SUBJECT_ID) REFERENCES TABLE_SUBJECT(SUBJECT_ID)   
  ENABLE 
);



CREATE TABLE TABLE_CURR_TEST 
(
  CURR_ID NUMBER(5) NOT NULL 
, TEST_ID NUMBER(5) NOT NULL 
, QUESTION_ID NUMBER(5) NOT NULL 
, USER_RESPONSE VARCHAR2(100) 
, RESPONSE_STATUS VARCHAR2(10) NOT NULL 
, CONSTRAINT TABLE_CURR_TEST_PK PRIMARY KEY 
  (
    CURR_ID 
  )
  , CONSTRAINT TABLE_CURR_TEST_TEST_FK FOREIGN KEY(TEST_ID) REFERENCES TABLE_TEST(TEST_ID)
  , CONSTRAINT TABLE_CURR_TEST_QUESTION_FK FOREIGN KEY(QUESTION_ID) REFERENCES TABLE_QUESTION(QUESTION_ID)
  ENABLE 
);



CREATE TABLE TABLE_USER_RESULT 
(
  USER_RESULT_ID NUMBER(5) NOT NULL 
, TEST_ID NUMBER(5) NOT NULL 
, SCORE NUMBER(5) NOT NULL 
, RESULT_STATUS VARCHAR2(20) NOT NULL 
, QUESTION_LEVEL NUMBER(1) NOT NULL 
, CONSTRAINT TABLE_USER_RESULTS_PK PRIMARY KEY 
  (
    USER_RESULT_ID 
  )
  , CONSTRAINT TABLE_USER_RESULT_TEST_FK FOREIGN KEY(TEST_ID) REFERENCES TABLE_TEST(TEST_ID)
  ENABLE
